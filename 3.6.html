class Student {
    String nume;

    Student(String nume) {
        this.nume = nume;
    }
}

public class Main {

    // Metodă care încearcă să modifice o valoare primitivă
    static void modificaPrimitiv(int x) {
        x = 10; // Modifică doar copia locală a lui x
    }

    // Metodă care modifică un obiect prin referință
    static void modificaObiect(Student s) {
        s.nume = "Alex"; // Modifică conținutul obiectului original
    }

    // Metodă care schimbă referința unui obiect
    static void schimbaReferinta(Student s) {
        s = new Student("Maria"); // Crează un nou obiect și schimbă referința locală
    }

    public static void main(String[] args) {
        int a = 5;
        modificaPrimitiv(a);
        System.out.println("Valoarea lui a după apel: " + a); // 5 (nemodificat)

        Student stud = new Student("Ion");
        modificaObiect(stud);
        System.out.println("Numele studentului după modificare: " + stud.nume); // "Alex" (modificat)

        schimbaReferinta(stud);
        System.out.println("Numele studentului după schimbare referință: " + stud.nume); // "Alex" (nemodificat)
    }
}
